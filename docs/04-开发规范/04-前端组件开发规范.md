# 前端组件开发规范

## 1. Vue 组件开发规范

### 1.1 组件文件结构

```vue
<template>
  <!-- 模板部分 -->
</template>

<script setup lang="ts">
// 逻辑部分
</script>

<style lang="less" scoped>
// 样式部分
</style>
```

### 1.2 组件命名

- 组件名: 大驼峰 PascalCase，如 `UserTable.vue`, `RoleForm.vue`
- 文件路径: 与路由/功能对应，如 `views/sys/user/list.vue`
- 通用组件: 以 `Basic` / `App` 开头，如 `BasicTable.vue`, `AppModal.vue`

### 1.3 Props 定义

```typescript
// ✅ 正确: 类型明确，有默认值
defineProps<{
  title: string;
  columns: BasicColumn[];
  pageSize?: number;        // 可选
  showSearch?: boolean;     // 可选
  loading?: boolean;        // 可选
}>();

// ❌ 避免: 广泛类型、无默认值
defineProps({
  title: { type: String, required: true },
  columns: { type: Array, required: true },
});
```

### 1.4 Emits 定义

```typescript
// ✅ 正确: 明确事件名称和载荷
const emit = defineEmits<{
  'update:visible': [value: boolean];
  submit: [data: Record<string, any>];
  cancel: [];
}>();
```

### 1.5 组件命名约定 (JeeSite 继承)

| 组件类别 | 命名 | 示例 |
|---|---|---|
| 表格 | BasicTable | `BasicTable` |
| 表单 | BasicForm | `BasicForm` |
| 树选择 | TreeSelect | `TreeSelect` |
| 弹窗 | BasicModal | `BasicModal` |
| 搜索 | SearchForm | `SearchForm` |
| 图标 | IconPicker | `IconPicker` |

## 2. TypeScript 类型定义

```typescript
// 类型定义文件 *.d.ts 或 *.type.ts

// 统一响应
interface ApiResult<T = any> {
  code: number;
  message: string;
  data: T;
}

// 分页请求
interface PageRequest {
  pageNo: number;
  pageSize: number;
  sortField?: string;
  sortOrder?: 'asc' | 'desc';
}

// 分页响应
interface PageResult<T> {
  list: T[];
  total: number;
  pageNo: number;
  pageSize: number;
}

// 用户类型
interface UserType {
  userCode: string;
  userName: string;
  loginCode: string;
  email?: string;
  phone?: string;
  status: string;
  orgName?: string;
}

// 树节点类型
interface TreeNode {
  code: string;
  name: string;
  parentCode: string;
  children?: TreeNode[];
}
```

## 3. Hooks 命名规范

```typescript
// ✅ 正确: useXxx 命名
useTable()       // 表格操作
useForm()        // 表单操作
useModal()       // 弹窗
usePermission()  // 权限检查
useDict()        // 字典数据
useCache()       // 缓存

// 使用示例
const [registerTable, { reload, setLoading }] = useTable();
const [registerForm, { resetFields, validate }] = useForm();
```

## 4. API 调用规范

```typescript
// src/api/sys/user.ts
import { request } from '@/utils/request';

export function fetchUserList(params: PageRequest) {
  return request.post<PageResult<UserType>>('/api/v1/sys/user/list', params);
}

export function fetchUser(params: { userCode: string }) {
  return request.get<UserType>('/api/v1/sys/user/get', { params });
}

export function saveUser(data: Record<string, any>) {
  return request.post('/api/v1/sys/user/save', data);
}

export function deleteUser(data: { userCode: string }) {
  return request.post('/api/v1/sys/user/delete', data);
}
```

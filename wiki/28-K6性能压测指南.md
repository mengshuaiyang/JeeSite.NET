# K6 性能压测指南

## 一、K6 简介

K6（由 Grafana Labs 维护）是基于 Go + JavaScript 的现代负载测试工具。

**核心特性**：
- 脚本化（JavaScript / ES2015-2022）
- CLI + 本地运行
- 可视化结果输出（JSON / CSV / Grafana Cloud）

**在 JeeSite.NET 中的用途**：
- API 接口压力测试（用户 / 文章 / 字典等接口）
- 首页 / 列表页并发性能测试
- 数据库连接池与缓存命中率评估
- 生产环境发布前的回归性能验证

## 四、基本脚本结构（scenario.js）

默认脚本内容：

```javascript
import http from 'k6/http';
import { check, sleep, group, fail } from 'k6';
import { Rate, Trend } from 'k6/metrics';
import { randomString, randomIntBetween, uuidv4 } from 'https://jslib.k6.io/k6-utils/1.4.0/index.js';

// ===== 自定义指标 =====
let errorRate = new Rate('error_rate');
let authTrend = new Trend('auth_response_time');
let articleListTrend = new Trend('article_list_time');
let searchTrend = new Trend('search_time');

// ===== 测试配置（VUs 数量 + 持续时间 + 阈值） =====
export const options = {
    stages: [
        { duration: '1m', target: 50 },     // 1 分钟内线性增加到 50 VUs
        { duration: '3m', target: 50 },     // 保持 50 并发 3 分钟
        { duration: '1m', target: 200 },    // 增加到 200 并发
        { duration: '3m', target: 200 },    // 保持 200 并发
        { duration: '1m', target: 0 }       // 平滑下降
    ],
    thresholds: {
        http_req_duration: ['p(95)<500'],   // 95% 请求在 500ms 内
        http_req_failed: ['rate<0.05'],     // 错误率低于 5%
        error_rate: ['rate<0.1'],           // 业务逻辑错误率 < 10%
        auth_response_time: ['p(95)<800'],  // 登录接口 95% 分位 < 800ms
        article_list_time: ['p(95)<600'],   // 文章列表 95% < 600ms
        search_time: ['p(95)<1200']         // 全文搜索 95% < 1.2s
    },
    // 若要运行分布式测试: k6 cloud scenario.js
    cloud: {
        name: 'JeeSite.NET Stress Test',
        projectID: 12345
    }
};

// ===== 环境配置 =====
const BASE_URL = __ENV.BASE_URL || 'http://localhost:5000';
const API_PREFIX = '/api/v1';

// ===== 登录: 每个 VU 仅在 setup 中执行一次 =====
// 注意: setup 阶段的返回值会在整个 VU 生命周期内共享
export function setup() {
    const loginRes = http.post(
        `${BASE_URL}${API_PREFIX}/sys/auth/login`,
        JSON.stringify({ loginCode: 'admin', password: 'admin' }),
        { headers: { 'Content-Type': 'application/json' } }
    );
    if (loginRes.status !== 200) {
        fail('Setup: 登录失败');
    }
    const body = JSON.parse(loginRes.body);
    if (body.code !== 0 || !body.data || !body.data.token) {
        fail('Setup: 获取 Token 失败');
    }
    return { token: body.data.token };
}

// ===== 主测试函数（每个 VU 每迭代执行一次） =====
export default function (data) {
    const headers = {
        'Content-Type': 'application/json',
        'Authorization': `Bearer ${data.token}`
    };

    // 场景 1: 用户列表查询（带分页和过滤）
    group('用户管理接口', function () {
        const page = randomIntBetween(1, 10);
        const pageSize = 20;
        const keyword = ['admin', 'test', 'user'][randomIntBetween(0, 2)];
        const res = http.get(
            `${BASE_URL}${API_PREFIX}/sys/user?pageIndex=${page}&pageSize=${pageSize}&keyword=${keyword}`,
            { headers: headers }
        );
        check(res, {
            '用户列表状态为 200': (r) => r.status === 200,
            '用户列表有返回数据': (r) => {
                const body = JSON.parse(r.body);
                return body.code === 0 && body.data && body.data.items && body.data.items.length > 0;
            }
        });
        errorRate.add(res.status !== 200);
        sleep(Math.random() * 2 + 1); // 1-3 秒随机等待，模拟真实用户
    });

    // 场景 2: CMS 文章列表
    group('CMS 文章接口', function () {
        const res = http.get(
            `${BASE_URL}${API_PREFIX}/cms/article?pageSize=20&pageIndex=1&isPublished=1`,
            { headers: headers }
        );
        const ok = check(res, {
            '文章列表状态为 200': (r) => r.status === 200,
            '文章列表格式正确': (r) => {
                const body = JSON.parse(r.body);
                return body.code === 0 && body.data && body.data.total >= 0;
            }
        });
        articleListTrend.add(res.timings.duration);
        errorRate.add(!ok);
        sleep(1);
    });

    // 场景 3: 全文搜索
    group('全文搜索接口', function () {
        const keywords = ['系统', '管理', '测试', '用户', '文章'];
        const kw = keywords[randomIntBetween(0, keywords.length - 1)];
        const res = http.get(
            `${BASE_URL}${API_PREFIX}/sys/search?q=${encodeURIComponent(kw)}&pageSize=10`,
            { headers: headers }
        );
        const ok = check(res, {
            '搜索状态为 200': (r) => r.status === 200,
            '搜索返回 JSON': (r) => {
                const body = JSON.parse(r.body);
                return body.code === 0;
            }
        });
        searchTrend.add(res.timings.duration);
        errorRate.add(!ok);
        sleep(1);
    });

    // 场景 4: 字典数据查询（高频率读接口）
    group('字典数据查询', function () {
        const res = http.get(
            `${BASE_URL}${API_PREFIX}/sys/dict-data/by-type/sys_user_status`,
            { headers: headers }
        );
        check(res, { '字典查询状态 200': (r) => r.status === 200 });
        sleep(0.5);
    });
}

// ===== 每个 VU 退出时清理 =====
export function teardown(data) {
    // 可选: 调用登出接口
    http.post(`${BASE_URL}${API_PREFIX}/sys/auth/logout`, null, {
        headers: { 'Authorization': `Bearer ${data.token}` }
    });
}
```

k6 run --vus 100 --duration 5m scenario.js

k6 run --vus 1000 --duration 10m scenario.js

## 七、常见压测场景

### 场景 A: 登录接口压力测试 (endpoints/auth-login.js)

目的：验证登录接口在高并发下的表现（Token 生成、密码校验、缓存）

```javascript
import http from 'k6/http';
import { check, sleep } from 'k6';

export const options = {
    stages: [
        { duration: '30s', target: 50 },
        { duration: '2m', target: 200 },
        { duration: '30s', target: 0 }
    ],
    thresholds: {
        http_req_duration: ['p(95)<800'],
        http_req_failed: ['rate<0.05']
    }
};

const BASE_URL = __ENV.BASE_URL || 'http://localhost:5000';

export default function() {
    const res = http.post(`${BASE_URL}/api/v1/sys/auth/login`,
        JSON.stringify({ loginCode: 'admin', password: 'admin' }),
        { headers: { 'Content-Type': 'application/json' } }
    );
    check(res, { '登录成功': (r) => r.status === 200 });
    sleep(1);
}
```

### 场景 B: 文件上传接口 (endpoints/upload.js)

目的：测试大文件上传性能 + 文件签名校验开销

```javascript
import http from 'k6/http';
import { check, sleep } from 'k6';
import { SharedArray } from 'k6/data';
import { open } from 'k6/experimental/fs';
import { randomIntBetween } from 'https://jslib.k6.io/k6-utils/1.4.0/index.js';

const BASE_URL = __ENV.BASE_URL || 'http://localhost:5000';
const TOKEN = __ENV.TOKEN || '';

const files = new SharedArray('test-files', function () {
    return [
        { name: 'small.png', path: 'fixtures/small.png', size: 50 * 1024 },      // 50KB
        { name: 'medium.jpg', path: 'fixtures/medium.jpg', size: 1 * 1024 * 1024 }, // 1MB
        { name: 'large.pdf', path: 'fixtures/large.pdf', size: 5 * 1024 * 1024 }   // 5MB
    ];
});

export default function() {
    const file = files[randomIntBetween(0, files.length - 1)];
    const binFile = open(file.path, 'b');
    const data = { file: binFile };
    const res = http.post(`${BASE_URL}/api/v1/sys/file/upload`, data, {
        headers: { 'Authorization': 'Bearer ' + TOKEN }
    });
    check(res, { '上传成功': (r) => r.status === 200 });
    sleep(2);
}
```

## 九、压测前准备检查清单

- [ ] 后端服务已启动（k6 run 前 `dotnet run --project src/JeeSiteNET.Web.Api`）
- [ ] Redis 服务正常（缓存命中率影响结果）
- [ ] SQL Server 连接正常（未在维护模式）
- [ ] Elasticsearch 索引已预热（避免首波请求冷启动）
- [ ] MinIO / 对象存储服务可访问（若测试文件上传）
- [ ] 前端服务未在 vite dev 模式（应使用生产构建）
- [ ] 测试环境与生产环境配置一致（appsettings.*.json）
- [ ] 已备份测试数据库（避免压测写入污染真实数据）
- [ ] 压测机与被测服务器网络畅通（1Gbps+ 内网）
- [ ] 当前无其他大规模任务（备份 / 重建索引 / ETL 等）

## 十、性能 SLO（服务等级目标）建议

| 接口 | p(95) 响应时间 | p(99) | 错误率 | 测试通过条件 |
|------|---------------|--------|---------|-------------|
| 登录 POST /auth/login | < 800ms | < 1.5s | < 2% | 200 VUs 持续 5 分钟达标 |
| 用户列表 GET /user?.. | < 500ms | < 1.0s | < 1% | 同上 |
| 文章列表 GET /cms/article | < 600ms | < 1.2s | < 1% | 同上 |
| 文章详情 GET /cms/article/{id} | < 300ms | < 500ms | < 1% | 同上 |
| 全文搜索 GET /search?q=xxx | < 1.2s | < 2.0s | < 3% | 同上 |
| 文件上传 POST /file/upload | < 2.0s（5MB） | < 5.0s | < 2% | 同上 |
| 验证码 POST /valid-code/send | < 500ms | < 1.0s | < 2% | 同上 |

## 十一、压测后分析与优化

1. 查看 `http_req_duration` 是否达标
2. 若 p(95) 超 SLO，查看 SQL Server Profiler 或 EF Core 日志中的慢查询
3. 检查 FusionCache 命中率（通过 `GET /api/v1/sys/cache/list` 查看）
4. 检查 Redis 连接数 & 延迟（`redis-cli info stats`）
5. 检查 SQL Server CPU / 内存 / 连接池使用
6. **常见优化方向**：
   - 慢查询 → 添加索引 / 重写查询
   - 高缓存 miss → 增大缓存过期时间 / 预热缓存
   - 文件上传慢 → 启用分片上传 / 对象存储 CDN
   - 登录接口高延迟 → 检查密码哈希算法是否足够快 / Redis 缓存 Token
   - 接口超时 → 检查数据库连接池大小（默认通常 100，可增至 200-500）

## 十二、故障排查

| 问题 | 排查方向 |
|------|---------|
| k6: Get "http://localhost:5000/api/v1/sys/auth/login" → dial tcp connect: connection refused | 后端服务未启动，检查端口 5000 是否被占用 |
| 所有请求返回 401 Unauthorized | Token 生成 / 传递方式错误，检查 `setup()` 函数 |
| http_req_duration p(95) 异常高 | 查看后端日志中慢请求的 SQL，可能缺少索引或 SQL 未优化 |
| http_req_failed 高错误率 | 检查接口业务逻辑，可能是高并发下的竞争条件（race condition） |
| 压测机 CPU 100% | K6 本身也需要 CPU，使用多机分布式测试（k6 cloud / k6 operator） |
| 内存持续增长（被测服务）| 可能有内存泄漏，使用 dotnet-dump 分析或启动时添加 GC 诊断 |

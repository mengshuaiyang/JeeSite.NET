## 一、测试架构总览

JeeSite.NET 的测试体系遵循「分层测试金字塔」策略：工具类单元测试 → 基础设施测试 → 业务服务测试 → API Controller 集成测试 → 前端组件测试 → E2E 端到端测试。

| 前端组件 | `Login.vue` `VditorEditor.vue` | 单元测试 | Vitest + Vue Test Utils | **>= 50%** |

## 三、工具类单元测试示例（EncryptUtil）

工具类是最适合单元测试的对象：无外部依赖、输入输出纯函数关系。

```csharp
using Xunit;
using FluentAssertions;
using JeeSiteNET.Core.Utils;

namespace JeeSiteNET.Tests.Core;

/// <summary>
/// EncryptUtil 加密工具类测试
/// 覆盖：MD5 / SHA1 / SHA256 / AES 加解密 / SM2 / SM4 / 随机密钥生成
/// </summary>
public class EncryptUtilTests
{
    // ─────────────────────────── MD5 ───────────────────────────

    [Fact]
    public void Md5_ShouldReturn32CharLowerHex_ForStandardInput()
    {
        // Arrange
        var input = "jeesite";

        // Act
        var result = EncryptUtil.Md5(input);

        // Assert
        result.Should().HaveLength(32);
        result.Should().MatchRegex("^[a-f0-9]+$");
        result.Should().Be("b333e16a1f6a6d11a2c6e32fb7f6d92a"); // 固定向量验证
    }

    [Theory]
    [InlineData("")]
    [InlineData(null)]
    public void Md5_ShouldReturnEmptyForNullOrEmptyInput(string? input)
    {
        var result = EncryptUtil.Md5(input!);
        result.Should().BeEmpty();
    }

    [Fact]
    public void Md5_ShouldBeCaseInsensitiveOnOutput()
    {
        // MD5 标准输出为小写十六进制
        var result = EncryptUtil.Md5("Test String");
        result.Should().Be(result.ToLowerInvariant());
    }

    // ─────────────────────────── SHA ───────────────────────────

    [Fact]
    public void Sha256_ShouldBeConsistentForSameInput()
    {
        var input = "重复相同输入测试一致性";
        var r1 = EncryptUtil.Sha256(input);
        var r2 = EncryptUtil.Sha256(input);

        r1.Should().Be(r2);
        r1.Should().HaveLength(64); // SHA-256 = 32 bytes hex
    }

    [Fact]
    public void Sha256_ShouldProduceDifferentOutputForSimilarInput()
    {
        // 雪崩效应验证：改动一个字符，输出完全不同
        var r1 = EncryptUtil.Sha256("Password123");
        var r2 = EncryptUtil.Sha256("Password124");

        r1.Should().NotBe(r2);
    }

    // ─────────────────────────── AES ───────────────────────────

    [Fact]
    public void AesEncryptThenDecrypt_ShouldReturnOriginalValue()
    {
        // Arrange
        var original = "测试明文 Test123!@# 特殊字符 &<>";
        var key = EncryptUtil.GenerateRandomKey(16);    // 128-bit key
        var iv = "1234567890abcdef";                      // 16-byte IV

        // Act
        var encrypted = EncryptUtil.EncryptAes(original, key, iv);
        var decrypted = EncryptUtil.DecryptAes(encrypted, key, iv);

        // Assert
        decrypted.Should().Be(original);
        encrypted.Should().NotBe(original);
        encrypted.Should().NotBeNullOrWhiteSpace();
    }

    [Fact]
    public void AesDecrypt_WithWrongKey_ShouldThrowOrProduceGarbage()
    {
        var original = "confidential data";
        var key1 = EncryptUtil.GenerateRandomKey(16);
        var key2 = EncryptUtil.GenerateRandomKey(16);
        var iv = "1234567890abcdef";

        var encrypted = EncryptUtil.EncryptAes(original, key1, iv);

        // 使用错误 key 解密
        var action = () => EncryptUtil.DecryptAes(encrypted, key2, iv);

        // AES-CBC 下错误 key 可能抛出异常或产生乱码
        // 本项目实现应当抛出 CryptographicException
        action.Should().Throw<System.Security.Cryptography.CryptographicException>();
    }

    [Fact]
    public void Aes_EmptyString_Roundtrip()
    {
        var key = EncryptUtil.GenerateRandomKey(16);
        var iv = "1234567890abcdef";

        var encrypted = EncryptUtil.EncryptAes("", key, iv);
        var decrypted = EncryptUtil.DecryptAes(encrypted, key, iv);

        decrypted.Should().Be("");
    }

    // ───────────────────── 随机密钥生成 ─────────────────────

    [Fact]
    public void GenerateRandomKey_ShouldProduceUniqueKeys()
    {
        var keys = new HashSet<string>();
        for (int i = 0; i < 1000; i++)
        {
            keys.Add(EncryptUtil.GenerateRandomKey(16));
        }
        keys.Count.Should().Be(1000); // 1000 次调用不应出现重复
    }

    [Theory]
    [InlineData(16)]
    [InlineData(24)]
    [InlineData(32)]
    public void GenerateRandomKey_ShouldRespectLength(int length)
    {
        var key = EncryptUtil.GenerateRandomKey(length);
        key.Should().HaveLength(length);
    }

    // ─────────────────────── 国密 SM4 ───────────────────────

    [Fact]
    public void Sm4_EncryptThenDecrypt_ShouldReturnOriginal()
    {
        var plaintext = "国密算法测试数据";
        var key = EncryptUtil.GenerateRandomKey(16); // SM4 key = 128-bit

        var encrypted = EncryptUtil.EncryptSm4(plaintext, key);
        var decrypted = EncryptUtil.DecryptSm4(encrypted, key);

        decrypted.Should().Be(plaintext);
    }
}
```

### 关键要点

1. **`[Fact]` vs `[Theory]`**：固定输入用 `Fact`，多组参数用 `Theory` + `InlineData`
2. **AAA 三段式**：Arrange（准备） → Act（执行） → Assert（断言）
3. **FluentAssertions 语法**：`result.Should().Be(...)` / `action.Should().Throw<...>()` / `result.Should().MatchRegex(...)`
4. **边界值**：空字符串、null、特殊字符、超长输入
5. **密码学额外要求**：固定向量测试（MD5 固定值）、雪崩效应（相似输入产生完全不同输出）、唯一性（随机密钥不重复）

## 四、仓储层集成测试（使用 In-Memory SQLite）

仓储层测试需要真实的数据库交互，但不能依赖 SQL Server 生产库。JeeSite.NET 使用 **SQLite In-Memory 模式**：每个测试类有独立的内存数据库，测试结束即销毁。

```csharp
using Xunit;
using FluentAssertions;
using Microsoft.EntityFrameworkCore;
using Microsoft.Data.Sqlite;
using JeeSiteNET.Infrastructure.Data;
using JeeSiteNET.Infrastructure.Repositories;
using JeeSiteNET.Modules.Sys.Domain.Entities;

namespace JeeSiteNET.Tests.Infrastructure;

/// <summary>
/// 所有仓储测试的基类：负责创建 SQLite In-Memory DbContext
/// </summary>
public class RepositoryTestBase : IDisposable
{
    protected readonly AppDbContext _db;
    protected readonly DbContextOptions<AppDbContext> _options;
    private readonly SqliteConnection _connection;

    public RepositoryTestBase()
    {
        // 使用共享内存连接（在同一连接对象生命周期内保持数据库存在）
        _connection = new SqliteConnection("DataSource=:memory:");
        _connection.Open();

        _options = new DbContextOptionsBuilder<AppDbContext>()
            .UseSqlite(_connection)
            .Options;

        _db = new AppDbContext(_options);
        _db.Database.EnsureCreated();   // 自动建表
        SeedTestData();                   // 可选：插入测试数据
    }

    protected virtual void SeedTestData()
    {
        // 可选：子类重写
    }

    public void Dispose()
    {
        _db.Dispose();
        _connection.Close();
        _connection.Dispose();
    }
}

/// <summary>
/// UserRepository 测试
/// </summary>
public class UserRepositoryTests : RepositoryTestBase
{
    protected override void SeedTestData()
    {
        _db.Users.Add(new User
        {
            Id = Guid.NewGuid().ToString(),
            UserCode = "admin",
            LoginCode = "admin",
            UserName = "超级管理员",
            Password = "hashed-pw",
            Status = 0,
            CreateDate = DateTimeOffset.UtcNow
        });
        _db.SaveChanges();
    }

    // ────── 查询 ──────

    [Fact]
    public async Task GetByCodeAsync_ExistingUser_ShouldReturnUser()
    {
        // Arrange
        var repo = new UserRepository(_db);

        // Act
        var result = await repo.GetByCodeAsync("admin");

        // Assert
        result.Should().NotBeNull();
        result.UserCode.Should().Be("admin");
        result.UserName.Should().Be("超级管理员");
    }

    [Fact]
    public async Task GetByCodeAsync_NonExistingUser_ShouldReturnNull()
    {
        var repo = new UserRepository(_db);
        var result = await repo.GetByCodeAsync("nonexistent");
        result.Should().BeNull();
    }

    [Fact]
    public async Task GetByCodeAsync_ShouldBeCaseInsensitive()
    {
        var repo = new UserRepository(_db);
        var result = await repo.GetByCodeAsync("ADMIN");

        result.Should().NotBeNull();
        result.UserCode.Should().Be("admin");
    }

    // ────── 新增 ──────

    [Fact]
    public async Task CreateAsync_ShouldAddNewUser()
    {
        var repo = new UserRepository(_db);
        var user = new User
        {
            Id = Guid.NewGuid().ToString(),
            UserCode = "test001",
            LoginCode = "test001",
            UserName = "测试用户",
            Password = "pw",
            Status = 0,
            CreateDate = DateTimeOffset.UtcNow
        };

        await repo.CreateAsync(user);

        var found = await _db.Users.FirstOrDefaultAsync(u => u.UserCode == "test001");
        found.Should().NotBeNull();
        found.UserName.Should().Be("测试用户");
    }

    // ────── 更新 ──────

    [Fact]
    public async Task UpdateAsync_ShouldModifyUser()
    {
        var repo = new UserRepository(_db);
        var user = await repo.GetByCodeAsync("admin");

        user.UserName = "修改后的管理员";
        await repo.UpdateAsync(user);

        var updated = await repo.GetByCodeAsync("admin");
        updated.UserName.Should().Be("修改后的管理员");
    }

    // ────── 删除 ──────

    [Fact]
    public async Task DeleteAsync_ShouldRemoveUser()
    {
        var repo = new UserRepository(_db);
        var user = await repo.GetByCodeAsync("admin");

        await repo.DeleteAsync(user);

        var found = await repo.GetByCodeAsync("admin");
        found.Should().BeNull();
    }

    // ────── 软删除 / 过滤 ──────

    [Fact]
    public async Task GetListAsync_ShouldNotReturnDeletedUsers()
    {
        // 假设 User 实体有 IsDeleted 字段，应用全局查询过滤器
        var repo = new UserRepository(_db);
        var user = await repo.GetByCodeAsync("admin");
        user.IsDeleted = 1;
        await repo.UpdateAsync(user);

        var list = await repo.GetListAsync();

        list.Should().NotContain(u => u.UserCode == "admin");
    }
}
```

### 为什么用 SQLite In-Memory？

| 方案 | 优点 | 缺点 |
|------|------|------|
| **SQLite In-Memory** | 独立、快速、零配置，每个测试类独立数据库 | SQL 方言与 SQL Server 略有差异 |
| 真实 SQL Server 实例 | 生产级一致性 | 需要数据库服务、测试间清理复杂、CI 集成成本高 |
| EF Core InMemory Provider | 纯内存、最快 | **不是真正的关系数据库**，不支持事务/约束/原始 SQL，容易掩盖 Bug |

> 🚫 **强烈不推荐使用 EF Core `UseInMemoryDatabase()`**。它不执行真正的关系数据库语义——忽略外键约束、不支持 `FromSqlRaw`、`TransactionScope` 无效。应使用 `UseSqlite("DataSource=:memory:")`。

## 五、业务服务层测试示例（ValidCodeService）

服务层测试的核心是**将外部依赖（数据库、缓存、邮件服务）全部 Mock**，只测业务逻辑本身。

```csharp
using Xunit;
using FluentAssertions;
using NSubstitute;
using Microsoft.Extensions.Caching.Distributed;
using Microsoft.Extensions.Logging;
using JeeSiteNET.Modules.Sys.Application.Services;
using JeeSiteNET.Modules.Sys.Application.Dtos;

namespace JeeSiteNET.Tests.Modules.Sys;

/// <summary>
/// 短信/邮件验证码服务测试
/// 外部依赖：IDistributedCache（存储验证码）、ISmsSender（发送短信）
/// </summary>
public class ValidCodeServiceTests
{
    private readonly IDistributedCache _cache;
    private readonly ILogger<ValidCodeService> _logger;
    private readonly ValidCodeService _service;

    public ValidCodeServiceTests()
    {
        // 每个测试方法独立创建 Mock，避免测试间状态污染
        _cache = Substitute.For<IDistributedCache>();
        _logger = Substitute.For<ILogger<ValidCodeService>>();
        _service = new ValidCodeService(_cache, _logger);
    }

    // ─────────── 生成验证码 ───────────

    [Fact]
    public async Task GenerateAndSendAsync_ShouldReturn6DigitCode()
    {
        // Act
        var code = await _service.GenerateAndSendAsync("test@example.com", "login");

        // Assert
        code.Should().NotBeNullOrEmpty();
        code.Should().MatchRegex("^[0-9]{6}$");

        // 验证缓存被写入（IDistributedCache.SetAsync 被调用一次）
        await _cache.Received(1).SetAsync(
            Arg.Any<string>(),
            Arg.Is<byte[]>(b => b != null && b.Length > 0),
            Arg.Any<DistributedCacheEntryOptions>(),
            Arg.Any<CancellationToken>());
    }

    [Fact]
    public async Task GenerateAndSendAsync_ShouldStoreCodeWithExpiry()
    {
        // Act
        await _service.GenerateAndSendAsync("13800138000", "login");

        // Assert：缓存选项中 TTL 应为 5 分钟
        await _cache.Received().SetAsync(
            Arg.Any<string>(),
            Arg.Any<byte[]>(),
            Arg.Is<DistributedCacheEntryOptions>(
                opts => opts.AbsoluteExpirationRelativeToNow.HasValue
                     && opts.AbsoluteExpirationRelativeToNow.Value <= TimeSpan.FromMinutes(5)),
            Arg.Any<CancellationToken>());
    }

    // ─────────── 验证码校验 ───────────

    [Fact]
    public async Task VerifyAsync_CorrectCode_ShouldReturnTrue()
    {
        // Arrange：准备一个真实的 code 存入缓存
        const string target = "13800138000";
        const string scene = "login";
        const string correctCode = "123456";

        var cacheKey = $"validcode:{scene}:{target}";
        _cache.GetAsync(cacheKey, Arg.Any<CancellationToken>())
            .Returns(Task.FromResult<byte[]?>(
                System.Text.Encoding.UTF8.GetBytes(correctCode)));

        // Act
        var ok = await _service.VerifyAsync(target, scene, correctCode);

        // Assert
        ok.Should().BeTrue();
    }

    [Fact]
    public async Task VerifyAsync_WrongCode_ShouldReturnFalse()
    {
        // Arrange
        const string target = "13800138000";
        _cache.GetAsync(Arg.Any<string>(), Arg.Any<CancellationToken>())
            .Returns(Task.FromResult<byte[]?>(
                System.Text.Encoding.UTF8.GetBytes("123456")));

        // Act
        var ok = await _service.VerifyAsync(target, "login", "000000");

        // Assert
        ok.Should().BeFalse();
    }

    [Fact]
    public async Task VerifyAsync_AfterExpiry_ShouldReturnFalse()
    {
        // 模拟验证码过期：GetAsync 返回 null
        _cache.GetAsync(Arg.Any<string>(), Arg.Any<CancellationToken>())
            .Returns(Task.FromResult<byte[]?>(null));

        var ok = await _service.VerifyAsync("expired@test.com", "login", "123456");

        ok.Should().BeFalse();
    }

    [Fact]
    public async Task VerifyAsync_ShouldBeCaseInsensitive()
    {
        // 邮件验证码可能大小写混合
        _cache.GetAsync(Arg.Any<string>(), Arg.Any<CancellationToken>())
            .Returns(Task.FromResult<byte[]?>(
                System.Text.Encoding.UTF8.GetBytes("AB3X9Y")));

        var ok = await _service.VerifyAsync("user@test.com", "email", "ab3x9y");

        ok.Should().BeTrue();
    }

    // ─────────── 输入校验 ───────────

    [Theory]
    [InlineData("")]
    [InlineData(null)]
    [InlineData("   ")]
    public async Task GenerateAndSendAsync_InvalidTarget_ShouldThrow(string? target)
    {
        var action = () => _service.GenerateAndSendAsync(target!, "login");
        await action.Should().ThrowAsync<ArgumentException>();
    }

    [Fact]
    public async Task GenerateAndSendAsync_RateLimit_ShouldBlockFrequentCalls()
    {
        // 同一目标 60 秒内不应发送第二次
        var target = "13800138000";
        await _service.GenerateAndSendAsync(target, "login");

        // 第二次调用应返回 null 或抛出异常（取决于实现）
        var action = () => _service.GenerateAndSendAsync(target, "login");

        // 这里的具体断言取决于服务实现：
        // 本项目实现应该抛出 UserFriendlyException("验证码发送过于频繁")
        await action.Should().ThrowAsync<JeeSiteNET.Core.Exceptions.UserFriendlyException>();
    }
}
```

### Mock 的关键技巧

1. **`Substitute.For<TInterface>()`**：创建接口的动态代理
2. **`.Returns(...)`**：设置方法返回值
3. **`.Received(1).Method(...)`**：断言方法被调用了 N 次
4. **`Arg.Any<T>()` / `Arg.Is<T>(predicate)`**：参数匹配（不必精确指定具体值，只需验证参数满足某条件）
5. **每个测试类独立创建 Substitute 实例**：避免静态共享导致测试间数据泄漏

## 六、Controller 层集成测试（TestServer）

Controller 层测试使用 **ASP.NET Core TestServer** + `HttpClient`：在测试进程内启动一个完整的 Web 服务器，加载真实的 Startup / Program 配置，通过 HTTP 发送请求并断言响应。

```csharp
using Microsoft.AspNetCore.Mvc.Testing;
using Microsoft.Extensions.DependencyInjection;
using System.Net.Http.Json;
using System.Text;
using JeeSiteNET;                                // 指向 Program.cs 所在命名空间
using JeeSiteNET.Modules.Sys.Application.Dtos;

namespace JeeSiteNET.Tests.Web;

/// <summary>
/// Controller 测试的基类：负责创建 TestServer 和 HttpClient
/// </summary>
public class ControllerTestBase : IClassFixture<WebApplicationFactory<Program>>
{
    protected readonly WebApplicationFactory<Program> _factory;
    protected readonly HttpClient _client;

    {

            builder.UseSetting("Environment", "Test");

            {
                // services.AddSingleton<ISmsSender, MockSmsSender>();
            });

/// <summary>
/// AuthController 测试
/// </summary>
public class AuthControllerTests : ControllerTestBase
{
    public AuthControllerTests(WebApplicationFactory<Program> factory) : base(factory) { }

    [Fact]

    {

        var request = new LoginRequest
        {
            LoginCode = "admin",
            Password = "admin"
        };

        // Act
        var response = await _client.PostAsJsonAsync("/api/v1/sys/auth/login", request);

        // Assert
        response.EnsureSuccessStatusCode();
        var result = await response.Content.ReadFromJsonAsync<ApiResult<LoginResultDto>>();

        result.Should().NotBeNull();
        result.Code.Should().Be(0);
        result.Data.Should().NotBeNull();
        result.Data.Token.Should().NotBeNullOrWhiteSpace();
        result.Data.UserInfo.Should().NotBeNull();
        result.Data.UserInfo.LoginCode.Should().Be("admin");
    }

    [Fact]
    public async Task Login_WrongPassword_ReturnsUnauthorized()
    {
        var response = await _client.PostAsJsonAsync(
            "/api/v1/sys/auth/login",
            new LoginRequest { LoginCode = "admin", Password = "wrong-password" });

        response.StatusCode.Should().Be(System.Net.HttpStatusCode.Unauthorized);
    }

    [Fact]
    public async Task Login_MissingFields_ReturnsBadRequest()
    {
        var json = new StringContent("{}", Encoding.UTF8, "application/json");
        var response = await _client.PostAsync("/api/v1/sys/auth/login", json);

        response.StatusCode.Should().Be(System.Net.HttpStatusCode.BadRequest);
    }

    [Fact]
    public async Task Me_WithoutToken_ReturnsUnauthorized()
    {
        var response = await _client.GetAsync("/api/v1/sys/auth/me");
        response.StatusCode.Should().Be(System.Net.HttpStatusCode.Unauthorized);
    }

    [Fact]
    public async Task Me_WithValidToken_ReturnsUserProfile()
    {
        // Step 1: 登录获取 Token
        var loginResponse = await _client.PostAsJsonAsync(
            "/api/v1/sys/auth/login",
            new LoginRequest { LoginCode = "admin", Password = "admin" });

        var loginResult = await loginResponse.Content.ReadFromJsonAsync<ApiResult<LoginResultDto>>();
        var token = loginResult.Data.Token;

        // Step 2: 携带 Token 请求 /me
        var request = new HttpRequestMessage(HttpMethod.Get, "/api/v1/sys/auth/me");

            new System.Net.Http.Headers.AuthenticationHeaderValue("Bearer", token);

        var response = await _client.SendAsync(request);

        // Assert
        response.EnsureSuccessStatusCode();
        var profile = await response.Content.ReadFromJsonAsync<ApiResult<UserProfileDto>>();
        profile.Data.Should().NotBeNull();
        profile.Data.LoginCode.Should().Be("admin");
    }
}
```

### TestServer 的特点

1. **真正的 HTTP 管道**：路由、Model Binding、中间件（CORS、鉴权、异常处理）、过滤器、Controller、格式化器全部参与执行
2. **无需端口占用**：在内存中运行，通过自定义 HttpMessageHandler 通信
3. **可替换依赖**：通过 `ConfigureTestServices` 替换 DI 容器中的服务（例如将生产的邮件服务替换为 Mock）
4. **`WebApplicationFactory<Program>`**：泛型参数指向主项目的 `Program` 类（C# 10 顶级语句下，需要 `Program` 类是 public partial，或用 `InternalsVisibleTo`）

在测试项目的 `.csproj` 中添加：

## 九、测试覆盖率目标建议

| 层级 | 建议覆盖率 | 重点覆盖内容 |
|------|-----------|-------------|
| 工具类 (Core/Utils) | **>= 90%** | 加密算法、字符串处理、格式校验、HTML 清洗白名单 |
| 基础设施层 | **>= 70%** | 仓储 CRUD、EF Core 查询、缓存策略、DB 迁移脚本 |
| 业务服务层 | **>= 80%** | 核心业务逻辑、权限校验、异常分支、状态机转换 |
| API Controller 层 | **>= 60%** | 请求验证、响应格式、错误处理、鉴权流程 |
| 前端组件 | **>= 50%** | 关键表单、列表渲染、权限可见性、路由跳转 |

> 🎯 **覆盖率只是手段，不是目的**：100% 覆盖率并不代表没有 Bug。重点是覆盖异常分支、边界条件和关键业务逻辑，而不是为了数字去写低价值的「getter/setter 测试」。

## 十一、测试最佳实践

### 1. 命名规范：被测方法_场景_预期结果

```csharp
// ✅ 清晰描述：什么方法、什么场景下、应当产生什么结果
public async Task Login_ValidCredentials_ReturnsToken() { }
public async Task Login_WrongPassword_Returns401() { }
public void Md5_EmptyInput_ReturnsEmptyString() { }

// ❌ 难以理解测试在做什么
public void Test1() { }
public void TestLogin() { }
```

### 2. AAA 三段式：每个测试都按 Arrange / Act / Assert 组织

```csharp
[Fact]
public void Divide_ByZero_ThrowsDivideByZeroException()
{
    // Arrange —— 准备被测对象和输入
    var calculator = new Calculator();
    int a = 10, b = 0;

    // Act —— 执行被测操作（用 Action/Func 包装以便断言异常）
    Action act = () => calculator.Divide(a, b);

    // Assert —— 验证结果
    act.Should().Throw<DivideByZeroException>();
}
```

### 3. 每个测试只测一件事

一个测试方法只验证一个行为。如果一个测试失败，应当能直接定位到哪里出了问题。

```csharp
// ✅ 好：一个测试一个断言点
[Fact] public void Create_ShouldReturnEntityWithId() { }
[Fact] public void Create_ShouldPersistToDatabase() { }
[Fact] public void Create_DuplicateCode_ShouldThrow() { }

// ❌ 不好：一个测试断言三件事，失败时不知道是哪件
[Fact] public void Create_ShouldWork()
{
    // 同时断言 id 非空 / 存在于数据库 / 重复码抛异常 ...
}
```

### 4. 测试应独立、可重复、可顺序无关

- 每个测试方法之间**不共享任何状态**（不要用 `static` 字段在测试间传数据）
- 测试结果与运行顺序无关（不要依赖前一个测试写入的数据）
- 同一份输入永远产生相同输出（避免 `DateTime.Now` / `Random` / 外部网络调用等不确定性因素）

### 5. 使用可注入的时间服务，而不是 `DateTime.Now`

```csharp
// ❌ 测试无法控制时间
public bool IsExpired() => ExpireTime < DateTime.Now;

// ✅ 可测试
public interface IDateTimeProvider
{
    DateTime Now { get; }
}

public bool IsExpired(IDateTimeProvider timeProvider)
    => ExpireTime < timeProvider.Now;

// 测试中：
var fakeTime = Substitute.For<IDateTimeProvider>();
fakeTime.Now.Returns(new DateTime(2026, 6, 12));
```

### 6. 异常测试必须写

每个 `throw` 的分支都应有对应的测试：

```csharp
[Fact]
public void Create_NullUser_ThrowsArgumentNullException()
{
    var service = new UserService(_mockRepo);
    var action = () => service.CreateAsync(null!);
    action.Should().ThrowAsync<ArgumentNullException>();
}
```

### 7. 边界值测试：空字符串、null、0、负数、最大值

```csharp
[Theory]
[InlineData(0)]
[InlineData(-1)]
[InlineData(int.MinValue)]
public void Divide_NonPositiveDivisor_Throws(int divisor)
{
    // ...
}
```

### 8. 不要测试框架本身（只测你自己写的代码）

```csharp
// ❌ 不要测：EF Core 的 FirstOrDefault 能返回第一条记录（这是 Microsoft 负责的）
// ✅ 要测：你的 Repository 拼接的查询条件是否正确返回了你期待的实体
```

### 9. 使用 IClassFixture / CollectionFixture 共享昂贵资源

```csharp
// 适用于只需初始化一次的资源（例如数据库 schema、HttpClient 池）
[Collection("Database collection")]
public class UserRepositoryTests
{
    private readonly DatabaseFixture _db;
    public UserRepositoryTests(DatabaseFixture db) { _db = db; }
}

public class DatabaseFixture : IDisposable
{
    // 整个测试集合内共享同一个实例
}
```

### 10. Mock 不要过度使用

不要 Mock 一切。如果被测代码和它的依赖都是你自己团队写的，使用**真实实现**做集成测试通常比 Mock 更有价值。Mock 的适用场景：

- ✅ 外部系统（SMTP / SMS / 第三方 API / 文件系统 / 系统时钟）
- ✅ 状态不稳定的依赖（Random、当前用户）
- ✅ 需要精确控制返回值来触发特定分支（如「网络超时」场景）
- ❌ 不要 Mock 自己写的简单逻辑（例如把 `List<T>` Mock 成 `IEnumerable<T>`）

## 十二、测试数据清理

| 层级 | 清理策略 |
|------|---------|
| **SQLite In-Memory** | 每个测试类实例化时自动创建新库，`Dispose()` 中销毁，天然隔离 |
| **真实 SQL Server 测试** | 在 `RepositoryTestBase` 中使用 `DbContextTransaction`，每个测试结束后 `Rollback()`；或使用 `TransactionScope(TransactionScopeOption.Required)` |
| **Redis 缓存** | 使用独立的 Redis DB（如 DB 15），测试类 `Dispose` 中执行 `FLUSHDB` |
| **文件系统** | 使用 `Path.GetTempPath()` 创建临时目录，测试结束后 `Directory.Delete(..., recursive: true)` |
| **前端 localStorage** | `beforeEach` 中 `localStorage.clear()` |

### TransactionScope 示例

```csharp
// 适用于使用真实 SQL Server 的集成测试
[Fact]
public async Task CreateAsync_Isolated()
{
    using (var scope = new TransactionScope(
        TransactionScopeOption.Required,
        new TransactionOptions { IsolationLevel = IsolationLevel.ReadCommitted },
        TransactionScopeAsyncFlowOption.Enabled))
    {
        // 测试逻辑...
        // scope.Dispose() 时自动回滚，数据不会真正写入数据库
    }
}
```

## 十三、故障排查

| 问题 | 可能原因 | 排查方向 |
|------|---------|---------|
| **`dotnet test` 找不到任何测试** | 测试类非 public / 方法缺少 `[Fact]` 或 `[Theory]` / SDK 版本不匹配 | 检查测试类的访问修饰符；检查方法上的 Fact 特性；`dotnet --list-sdks` 确认版本 |
| **SQLite 内存库测试中表不存在** | `UseSqlite("DataSource=:memory:")` 必须保持同一个 `SqliteConnection` 对象 Open；一旦连接关闭，数据即丢失 | 确保 `RepositoryTestBase` 中创建的连接对象贯穿整个测试类，不要在测试方法内新建连接 |
| **`EnsureCreated()` 不迁移** | `EnsureCreated` 只创建 schema，**不执行 Migrations** | 如需测试迁移脚本，使用 `dbContext.Database.Migrate()` 替代 |
| **FluentAssertions 字符串断言失败但看起来一样** | 行尾换行符差异（`\r\n` vs `\n`）、不可见字符 / BOM | 使用 `.Should().BeEquivalentTo(..., opts => opts.WithStrictOrdering())`；或 `result.Should().Contain()` 先缩小范围 |
| **`NSubstitute` 收到调用但断言失败** | 参数匹配过于严格；或方法调用发生在尚未 Setup 的时机 | 改用 `Arg.Any<T>()` 放宽参数匹配；在断言前调用 `Received.InOrder()` 查看实际收到的调用序列 |
| **`TestServer` 返回 500 但本地运行正常** | TestServer 使用的 `appsettings.json` 与本地不同；或某些服务未在测试环境中注册 | 在测试类中 `factory.Server.Services.GetRequiredService<ILogger<...>>()` 收集日志；使用 `ConfigureTestServices` 补全依赖 |
| **前端 Vitest 中 DOM 不更新** | Vue 的响应式更新是异步的；操作后未 `await nextTick()` | 在每个 trigger / setProps / setValue 之后 `await nextTick()`，然后再断言 |
| **`@vue/test-utils` 中 Ant Design Vue 组件报错** | Ant Design Vue 依赖全局注册或特定 provide | 在 `setup.ts` 中全局注册 Ant Design Vue；或使用 `mount({ global: { plugins: [...] } })` |
| **覆盖率极低** | 实际测试文件很少；或 `include` 路径配置错误；或 TypeScript 源文件未被 instrumentation | 检查 `coverlet.msbuild` 是否正确安装；检查 `vitest.config.ts` 的 coverage.include；运行时加上 `-v normal` 查看每个测试是否被执行 |
| **CI 中测试通过但本地失败（或相反）** | 环境差异：操作系统路径分隔符、行尾符、时区、文化（CultureInfo）、环境变量 | 在本地使用 `dotnet test --filter ... -- RunConfiguration.TargetPlatform=x64` 对齐；在测试类构造函数中 `CultureInfo.CurrentCulture = CultureInfo.InvariantCulture` 强制中性文化 |
| **测试运行极慢** | 测试方法数量太多并行度不够；或测试间共享了重量级资源（例如每个测试类都启动一个 TestServer）；或代码中存在阻塞调用 | 使用 `dotnet test --parallel`；使用 `IClassFixture` / `CollectionDefinition` 共享 TestServer；检查测试中的 `Task.Wait()` / `.Result` 死锁 |

---

```csharp
using Xunit;
using NSubstitute;
using JeeSiteNET.Core.Utils;

namespace JeeSiteNET.Core.Tests;

public class EncryptUtilTests
{
    [Fact]
    public void Md5_ShouldReturn32CharHex_WhenInputIsString()
    {
        var result = EncryptUtil.Md5("hello");
        Assert.NotNull(result);
        Assert.Equal(32, result.Length);
        Assert.All(result, c => Assert.True(char.IsLetterOrDigit(c)));
    }

    [Theory]
    [InlineData("", "d41d8cd98f00b204e9800998ecf8427e")]
    [InlineData("a", "0cc175b9c0f1b6a831c399e269772661")]
    public void Md5_KnownValues_ReturnExpected(string input, string expected)
    {
        var result = EncryptUtil.Md5(input);
        Assert.Equal(expected, result, ignoreCase: true);
    }
}

// ===== 使用 NSubstitute 模拟依赖 =====
public class UserServiceTests
{
    [Fact]
    public async Task GetByIdAsync_WhenUserExists_ReturnsUser()
    {
        // Arrange
        var mockRepo = Substitute.For<IUserRepository>();
        var expectedUser = new User { Id = "1", LoginCode = "admin" };
        mockRepo.GetByIdAsync("1").Returns(Task.FromResult(expectedUser));
        var service = new UserService(mockRepo);

        // Act
        var result = await service.GetByIdAsync("1");

        // Assert
        Assert.NotNull(result);
        Assert.Equal("admin", result.LoginCode);
        await mockRepo.Received().GetByIdAsync("1");
    }
}
```

```csharp
// ===== 集成测试：WebApplicationFactory =====
using Microsoft.AspNetCore.Mvc.Testing;

public class AuthIntegrationTests : IClassFixture<WebApplicationFactory<Program>>
{
    private readonly HttpClient _client;

    public AuthIntegrationTests(WebApplicationFactory<Program> factory)
    {
        _client = factory.CreateClient();
    }

    public async Task Login_ValidCredentials_ReturnsToken()

        Assert.NotNull(body?.Token);
    }
}
```

## ❓ 常见问题（3-5 个）

1. **问：单元测试中如何处理数据库依赖？**  
   答：使用 NSubstitute 模拟 `IRepository` 接口，或使用 SQLite 内存数据库作为真实 DbContext 的底层。

2. **问：如何计算测试覆盖率？**  
   答：安装 `coverlet.collector` NuGet 包，运行 `dotnet test --collect:"XPlat Code Coverage"` 生成覆盖率报告。

3. **问：集成测试如何隔离数据库？**  
   答：为每个测试类创建独立的数据库（如 `test_{Guid}`），测试结束后自动清理，避免测试之间的数据污染。
